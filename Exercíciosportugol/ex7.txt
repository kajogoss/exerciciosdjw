notas: vetor [1..200] de inteiro
contador,n,moda: inteiro
	escreval("Digite o número de alunos: ")
	leia(n)

	para contador de 1 ate n faca
		escreva("Informe a",contador,"° nota: ")
		leia (notas[contador])
		moda <- 0
		se (notas[contador] > 1) então
			moda <- moda + 1
			notas[contador] <- moda
		fimse
	fimpara
